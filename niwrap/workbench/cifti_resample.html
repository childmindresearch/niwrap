550def cifti_resample(
551    cifti_in: InputPathType,
552    direction: str,
553    cifti_template: InputPathType,
554    template_direction: str,
555    surface_method: str,
556    volume_method: str,
557    cifti_out: str,
558    opt_surface_largest: bool = False,
559    volume_predilate: CiftiResampleVolumePredilate | None = None,
560    surface_postdilate: CiftiResampleSurfacePostdilate | None = None,
561    affine: CiftiResampleAffine | None = None,
562    warpfield: CiftiResampleWarpfield | None = None,
563    left_spheres: CiftiResampleLeftSpheres | None = None,
564    right_spheres: CiftiResampleRightSpheres | None = None,
565    cerebellum_spheres: CiftiResampleCerebellumSpheres | None = None,
566    runner: Runner | None = None,
567) -> CiftiResampleOutputs:
568    """
569    Resample a cifti file to a new cifti space.
570    
571    Resample cifti data to a different brainordinate space. Use COLUMN for the
572    direction to resample dscalar, dlabel, or dtseries. Resampling both
573    dimensions of a dconn requires running this command twice, once with COLUMN
574    and once with ROW. If you are resampling a dconn and your machine has a
575    large amount of memory, you might consider using
576    -cifti-resample-dconn-memory to avoid writing and rereading an intermediate
577    file. The <template-direction> argument should usually be COLUMN, as
578    dtseries, dscalar, and dlabel all have brainordinates on that direction. If
579    spheres are not specified for a surface structure which exists in the cifti
580    files, its data is copied without resampling or dilation. Dilation is done
581    with the 'nearest' method, and is done on <new-sphere> for surface data.
582    Volume components are padded before dilation so that dilation doesn't run
583    into the edge of the component bounding box. If neither -affine nor
584    -warpfield are specified, the identity transform is assumed for the volume
585    data.
586    
587    The recommended resampling methods are ADAP_BARY_AREA and CUBIC (cubic
588    spline), except for label data which should use ADAP_BARY_AREA and
589    ENCLOSING_VOXEL. Using ADAP_BARY_AREA requires specifying an area option to
590    each used -*-spheres option.
591    
592    The <volume-method> argument must be one of the following:
593    
594    CUBIC
595    ENCLOSING_VOXEL
596    TRILINEAR
597    
598    The <surface-method> argument must be one of the following:
599    
600    ADAP_BARY_AREA
601    BARYCENTRIC
602    .
603    
604    Author: Connectome Workbench Developers
605    
606    URL: https://github.com/Washington-University/workbench
607    
608    Args:
609        cifti_in: the cifti file to resample.
610        direction: the direction of the input that should be resampled, ROW or\
611            COLUMN.
612        cifti_template: a cifti file containing the cifti space to resample to.
613        template_direction: the direction of the template to use as the\
614            resampling space, ROW or COLUMN.
615        surface_method: specify a surface resampling method.
616        volume_method: specify a volume interpolation method.
617        cifti_out: the output cifti file.
618        opt_surface_largest: use largest weight instead of weighted average or\
619            popularity when doing surface resampling.
620        volume_predilate: dilate the volume components before resampling.
621        surface_postdilate: dilate the surface components after resampling.
622        affine: use an affine transformation on the volume components.
623        warpfield: use a warpfield on the volume components.
624        left_spheres: specify spheres for left surface resampling.
625        right_spheres: specify spheres for right surface resampling.
626        cerebellum_spheres: specify spheres for cerebellum surface resampling.
627        runner: Command runner.
628    Returns:
629        NamedTuple of outputs (described in `CiftiResampleOutputs`).
630    """
631    runner = runner or get_global_runner()
632    execution = runner.start_execution(CIFTI_RESAMPLE_METADATA)
633    cargs = []
634    cargs.append("wb_command")
635    cargs.append("-cifti-resample")
636    cargs.append(execution.input_file(cifti_in))
637    cargs.append(direction)
638    cargs.append(execution.input_file(cifti_template))
639    cargs.append(template_direction)
640    cargs.append(surface_method)
641    cargs.append(volume_method)
642    cargs.append(cifti_out)
643    if opt_surface_largest:
644        cargs.append("-surface-largest")
645    if volume_predilate is not None:
646        cargs.extend(volume_predilate.run(execution))
647    if surface_postdilate is not None:
648        cargs.extend(surface_postdilate.run(execution))
649    if affine is not None:
650        cargs.extend(affine.run(execution))
651    if warpfield is not None:
652        cargs.extend(warpfield.run(execution))
653    if left_spheres is not None:
654        cargs.extend(left_spheres.run(execution))
655    if right_spheres is not None:
656        cargs.extend(right_spheres.run(execution))
657    if cerebellum_spheres is not None:
658        cargs.extend(cerebellum_spheres.run(execution))
659    ret = CiftiResampleOutputs(
660        root=execution.output_file("."),
661        cifti_out=execution.output_file(cifti_out),
